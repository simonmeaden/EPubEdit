240 static QString
241 name(const QColor& color)
242 {
243  auto colorKeys = COLORMAP.keys();
244  for (auto &k : colorKeys) {
245  auto c = COLORMAP.value(k);
246  if (c == color) {
247  return k;
248  }
249  }
250  return QString();
